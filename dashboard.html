html`<div class="kwn-statement" id="kwn-stmt">${(current_statement || "Bitte eine These auswählen.").replace(/\{location\}/g, cities_sorted[0] ?? "der Stadt")}</div>`
{
  const tip = document.createElement("div");
  Object.assign(tip.style, {
    position: "fixed",
    background: "white",
    border: "1px solid #ccc",
    borderRadius: "6px",
    padding: "8px 12px",
    fontSize: "12px",
    lineHeight: "1.45",
    fontFamily: "system-ui, sans-serif",
    pointerEvents: "none",
    zIndex: "9999",
    maxWidth: "340px",
    whiteSpace: "normal",
    boxShadow: "0 2px 8px rgba(0,0,0,0.18)",
    display: "none"
  });
  document.body.appendChild(tip);
  invalidation.then(() => tip.remove());

  if (filtered.length === 0) {
    const msg = !selected_parties || selected_parties.length === 0
      ? "Bitte mindestens eine Partei auswählen."
      : "Für diese These liegen keine Daten vor.";
    return html`<div style="padding:2rem; color:#666; text-align:center; border:1px dashed #ddd; border-radius:6px;">${msg}</div>`;
  }

  const chart = Plot.plot({
    width: plot_width,
    height: Math.max(500, cities_sorted.length * 36 + 180),
    marginLeft: 200,
    marginRight: 80,
    marginTop: 70,
    marginBottom: 100,
    style: { fontSize: "14px" },
    x: {
      domain: [0.5, 5.5],
      ticks: [1, 2, 3, 4, 5],
      tickFormat: d => cat_labels[d - 1],
      label: null,
      grid: true
    },
    y: {
      domain: cities_sorted,
      label: null
    },
    color: {
      domain: Object.keys(party_color_map),
      range: Object.values(party_color_map),
      legend: true
    },
    marks: [
      Plot.ruleX([1, 2, 3, 4, 5], { stroke: "#eee" }),
      Plot.dot(filtered_jittered, {
        x: "x_jitter",
        y: "gmd_name",
        fill: "party_name",
        stroke: "#444",
        strokeWidth: 0.8,
        r: 7
      }),
      Plot.dot(filtered_jittered, Plot.pointer({
        x: "x_jitter",
        y: "gmd_name",
        fill: "party_name",
        stroke: "black",
        strokeWidth: 2,
        r: 9
      }))
    ]
  });

  const esc = s => String(s ?? "").replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");

  function updateStatement(city) {
    const el = document.getElementById("kwn-stmt");
    if (!el) return;
    const stmt = current_statement || "Bitte eine These auswählen.";
    el.textContent = stmt.replace(/\{location\}/g, city ?? cities_sorted[0] ?? "der Stadt");
  }

  chart.addEventListener("input", () => {
    const d = chart.value;
    updateStatement(d ? d.gmd_name : null);
    if (!d) { tip.style.display = "none"; return; }
    tip.innerHTML =
      `<div style="margin-bottom:6px; font-size:11px; color:#555; line-height:1.6">` +
        `<b>${esc(d.party_name)}</b> &middot; ${esc(d.gmd_name)} &middot; ${esc(d.cat_label)}` +
      `</div>` +
      `<div style="font-size:12px; line-height:1.55; color:${d.explanation ? "#222" : "#999"}; font-style:${d.explanation ? "normal" : "italic"}">${esc(d.explanation || "Diese Partei hat keinen Begründungstext abgegeben.")}</div>`;
    tip.style.display = "block";
  });

  chart.addEventListener("pointermove", (e) => {
    if (tip.style.display === "none") return;
    const gap = 14;
    const x = e.clientX + gap;
    const y = e.clientY + gap;
    const tw = tip.offsetWidth;
    const th = tip.offsetHeight;
    tip.style.left = (x + tw > window.innerWidth  ? e.clientX - tw - gap : x) + "px";
    tip.style.top  = (y + th > window.innerHeight ? e.clientY - th - gap : y) + "px";
  });

  chart.addEventListener("pointerleave", () => {
    tip.style.display = "none";
    updateStatement(null);
  });

  return chart;
}

data = transpose(raw_data)
politikfelder = Array.from(new Set(data.map(d => d.these_politikfeld))).sort()
party_order = ["CDU", "SPD", "AfD", "GRÜNE", "DIE LINKE", "FDP", "BSW", "VOLT", "Die PARTEI"]
available_parties = party_order.filter(p => data.some(d => d.party_name === p))
party_color_map = ({
  "CDU":        "#000000",
  "SPD":        "#E3000F",
  "AfD":        "#009EE0",
  "GRÜNE":      "#1AA037",
  "DIE LINKE":  "#BE3075",
  "FDP":        "#FFED00",
  "BSW":        "#8B008B",
  "VOLT":       "#61259e",
  "Die PARTEI": "#C9002B"
})
cat_labels = ["1 – keine Zustimmung", "2", "3 – neutral", "4", "5 – volle Zustimmung"]
val_to_cat = function(v) {
  if (v === 0)   return 1;
  if (v === 25)  return 2;
  if (v === 50)  return 3;
  if (v === 75)  return 4;
  if (v === 100) return 5;
  return null;
}
theses_for_field = Array.from(new Set(
  data.filter(d => d.these_politikfeld === selected_field)
      .map(d => d.these_title)
)).sort()
filtered = (selected_thesis && selected_parties && selected_parties.length > 0)
  ? data
      .filter(d => d.these_title === selected_thesis &&
                   selected_parties.includes(d.party_name))
      .map(d => {
        const c = val_to_cat(d.valuation);
        return { ...d, cat_num: c, cat_label: c !== null ? cat_labels[c - 1] : null };
      })
      .filter(d => d.cat_num !== null)
  : []
filtered_jittered = {
  const groups = d3.group(filtered, d => d.gmd_name + "|" + d.cat_num);
  const out = [];
  for (const [, items] of groups) {
    const n = items.length;
    items.forEach((it, i) => {
      const off = n === 1 ? 0 : -0.2 + (0.4 * i) / (n - 1);
      out.push({ ...it, x_jitter: it.cat_num + off });
    });
  }
  return out;
}
cities_sorted = Array.from(new Set(filtered.map(d => d.gmd_name))).sort()
current_statement = filtered.length > 0 ? filtered[0].statement : ""
// Misst die Breite des Main-Containers und aktualisiert reaktiv bei Resize
plot_width = {
  const el = document.querySelector(".kwn-main");
  if (!el) return 900;
  const initial = el.getBoundingClientRect().width;
  return Generators.observe(notify => {
    notify(Math.max(600, Math.floor(initial)));
    const ro = new ResizeObserver(entries => {
      for (const e of entries) {
        notify(Math.max(600, Math.floor(e.contentRect.width)));
      }
    });
    ro.observe(el);
    return () => ro.disconnect();
  });
}
